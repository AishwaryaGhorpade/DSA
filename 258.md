https://leetcode.com/problems/add-digits/
```java
class Solution {
    public int addDigits(int num) {
         if(num<10)
        {
            return num;
        }
        int ans=0;
        while(num>0)
        {
            int sum=0;
            while(num>0)
            {
                int digit=num%10;
                sum+=digit;
                num/=10;
            }
            if(sum>=10)
            {
                num=sum;
            }
            if(sum<10)
            {
                ans=sum;
            }
        }
       return ans;
    }
}
```
